<script>
const teams = {
"St Mary’s": { played: 0, won: 0, drawn: 0, lost: 0, points: 0 },
"Lochee": { played: 0, won: 0, drawn: 0, lost: 0, points: 0 },
"Stobie FC": { played: 0, won: 0, drawn: 0, lost: 0, points: 0 },
"Ardler FC": { played: 0, won: 0, drawn: 0, lost: 0, points: 0 },
"Fairmuir": { played: 0, won: 0, drawn: 0, lost: 0, points: 0 }
};

function updateTable() {
const tbody = document.querySelector("#leagueTable tbody");
tbody.innerHTML = "";

for (let team in teams) {
const row = document.createElement("tr");
const data = teams[team];

row.innerHTML = `
<td>${team}</td>
<td>${data.played}</td>
<td>${data.won}</td>
<td>${data.drawn}</td>
<td>${data.lost}</td>
<td>${data.points}</td>
`;
tbody.appendChild(row);
}
}

document.getElementById("matchForm").addEventListener("submit", function(e) {
e.preventDefault();
const teamA = document.getElementById("teamA").value;
const teamB = document.getElementById("teamB").value;
const scoreA = parseInt(document.getElementById("scoreA").value);
const scoreB = parseInt(document.getElementById("scoreB").value);

if (teamA === teamB) {
alert("A team cannot play against itself.");
return;
}

// Update played
teams[teamA].played++;
teams[teamB].played++;

if (scoreA > scoreB) {
teams[teamA].won++;
teams[teamB].lost++;
teams[teamA].points += 3;
} else if (scoreA < scoreB) {
teams[teamB].won++;
teams[teamA].lost++;
teams[teamB].points += 3;
} else {
teams[teamA].drawn++;
teams[teamB].drawn++;
teams[teamA].points += 1;
teams[teamB].points += 1;
}

updateTable();
this.reset();
});

updateTable(); // Initial table load
</script>
